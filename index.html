mist.js

A small global object of JavaScript utilities that I tend to need frequently in my projects. A collection of time-saving helpers and commonly used functions.

// easy object utilities

let obj = {a: 1, b: 2, c: 3}
let obj2 = {a: 1, b: 2, c: 3}

console.log(mist.isEqual(obj, obj2)) // true
console.log(mist.isEqual(obj, {a: 1, b: 2})) // false

console.log(mist.pick(obj, ["a", "b"])) // {a: 1, b: 2}
